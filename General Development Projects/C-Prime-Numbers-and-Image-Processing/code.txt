/*
    TODO: Checks if a number is prime or not.
    
    It RETURNS 1 if the number is prime, otherwise it returns 0.
*/
int isPrime(int number) {
  if (number == 1){
       return 0;
    }
   
    else if (number >=2){
    
     for (int num_btwn = 2; num_btwn < number; num_btwn++)
     {
          if(number % num_btwn == 0){
             return 0;
         }
     }
    }
    
    else {
        return 1;
	}
}


/*
    TODO: PRINTS the prime numbers from 1 to rangeEnd.

    Uses the function isPrime to check if each number is prime or not.
*/
void primesInRange(int rangeEnd) {
    int a = 1;
	
    for(int num_btwn = 2; num_btwn <= rangeEnd; num_btwn++)
	{
        if (isPrime(num_btwn))
        {
            if (a == 1) {
                printf("%d", num_btwn);
            }

            else {
                printf(", %d", num_btwn);
            }
            a++;
        }
       
	}
    printf("\n");
}
